namespace = flavor_gr 

country_event = { #Bandits! (on bi-yearly pulse 4)
    id = flavor_gr.1
    title = flavor_gr.1.t
    desc = flavor_gr.1.d
    picture = TRADEGOODS_EventPicture
    is_triggered_only = yes
    trigger = {
        any_owned_province = {
            NOT = {
                development = 10
            }
        }
        any_neighbor_country = {
            NOT = {
                is_subject = yes
            }
        }
        NOT = {
        has_country_modifier = had_bandits
        has_country_modifier = had_recent_bandits 
        }

    }
    immediate = {
        random_neighbor_country = {
            limit = {
            NOT = { is_subject = yes }
            }
            save_event_target_as = neighbor_country 
        }
        random_owned_province = {
            limit = {
                NOT = { development = 10 }
            }
            save_event_target_as = targeted_province
        }
        add_country_modifier = {
            name = had_recent_bandits
            duration = 90
            hidden = yes
        }
        add_country_modfier = {
            name = had_bandits
            duration = 1460
            hidden = yes 
        }
        event_target:neighbor_country = {
            set_country_flag = bandit_origin_country 
        }
    }
    option = { #They came from [neighbor_country.GetName], clearly this is all their fault. Demand compensation for the attacks.
        name = flavor_gr.1.a
        event_target:targeted_province = {
            add_devastation = 2
        }
        event_target:neighbor_country = {
            country_event = {
                id = flavor_gr.2
                days = 5
                random = 5
            }
        }
    }
    option = { #We need to deal with it ourselves.
        name = flavor_gr.1.b
        event_target:targeted_province = {
            add_devastation = 2
        }
        add_treasury = -100
        add_country_modifier = {
            name = bandits_dealt_with
            duration = 3650
        }
        custom_tooltip = bandits_dealt_with_tt         
    }
    option = { #It's really not a big deal. 
        name = flavor_gr.1.c 
        event_target:targeted_province = {
            add_devastation = 4
        }
        custom_tooltip = marauding_bandits_tt
        remove_country_modifier = had_recent_bandits 
    }
}
