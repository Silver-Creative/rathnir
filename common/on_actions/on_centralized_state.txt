#ROOT = province that was centralized in
on_centralized_state = {
	if = {
		limit = {
			owner = {
				has_government_attribute = centralize_state_no_reform_cost
			}
		}
		owner = {
			add_adm_power = 50
			change_government_reform_progress = 50
		}
	}
	else_if = {
		limit = {
			owner = {
				has_government_attribute = centralize_state_half_refund
			}
		}
		owner = {
			add_adm_power = 25
			change_government_reform_progress = 25
		}
	}
	if = {
		limit = {
			owner = {
				has_government_attribute = centralize_state_development
			}
		}
		area = {
			limit = {
				owned_by = ROOT
			}
			random_list = {
				1 = {
					add_base_tax = 1
				}
				1 = {
					add_base_production = 1
				}
				1 = {
					add_base_manpower = 1
				}
			}
		}
	}
}
